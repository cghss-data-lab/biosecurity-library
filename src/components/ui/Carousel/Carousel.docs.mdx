import Carousel from './Carousel'

This component provides the structure and animations for a carousel, which allows paging forward and back and jumping to any page using the dots above the carousel. The buttons and dots have customizable colors for active and inactive status, and if only the buttons are assigned colors, the dots will automatically match.

```jsx live=true
// prettier-ignore
() => {
  // any component can be used for the page, here's a simple div with some styles
  const style = { width: '200px', height: '100px', background: 'lightgrey', display: 'flex', justifyContent: 'center', alignItems: 'center', borderRadius: 5}
  const Page = ({children}) => <div style={style}>{children}</div>

  return (
    <Carousel>
      <Page>Page 1!</Page>
      <Page>Page 2!</Page>
      <Page>Page 3!</Page>
    </Carousel>
  )
}
```
